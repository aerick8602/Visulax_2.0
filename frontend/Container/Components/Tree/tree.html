// Pre-order traversal (root, left, right)
void preOrderTraversal(Node* root) {
  if (root == nullptr) {
    return;  // Base case: If the root is null, we've reached the end of a branch and can return.
  }

  std::cout << root->data << " ";  // Visit the root node first: Print the data.

  preOrderTraversal(root->left);   // Visit the left subtree.

  preOrderTraversal(root->right);  // Visit the right subtree.
}

// In-order traversal (left, root, right)
void inOrderTraversal(Node* root) {
  if (root == nullptr) {
    return;  // Base case: If the root is null, we've reached the end of a branch and can return.

  inOrderTraversal(root->left);    // Visit the left subtree first.

  std::cout << root->data << " ";  // Visit the root node: Print the data
  
  inOrderTraversal(root->right);   // Visit the right subtree after the root.
